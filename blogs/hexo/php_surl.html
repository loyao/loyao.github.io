function getResponse($url, $data = [], $ip,$cookie_file = '', $timeout = 3){
if (empty($cookie_file)) {
$cookie_file = '.cookie';
}

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_REFERER, "https://".$ip."/SASDriver");  //构造来路
curl_setopt($ch, CURLOPT_USERAGENT, "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/54.0.2840.59 Safari/537.36");
if (!empty($data)) {
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS,($data));
curl_setopt($ch, CURLOPT_REFERER, "https://".$ip."/SASLogon/login");  //构造来路
}

$Header = array('Sec-Fetch-Site: same-origin', 'Host: '.$ip,"Origin: https://".$ip,'Content-type: application/x-www-form-urlencoded', 'Content-length: 150');

curl_setopt($ch, CURLOPT_HTTPHEADER, $Header);//请求头
curl_setopt($ch, CURLOPT_TIMEOUT, 100); //设置curl执行超时时间最大是多少
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);//取消ssl证书验证
curl_setopt($ch, CURLOPT_COOKIEJAR, $cookie_file);// 取cookie的参数是
curl_setopt($ch, CURLOPT_COOKIEFILE, $cookie_file); //发送cookie
curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, $timeout);
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);//取消ssl证书验证
curl_setopt($ch, CURLOPT_HEADER, TRUE);    //表示需要response header
    curl_setopt($ch, CURLOPT_NOBODY, FALSE); //表示需要response body

try {
$handles = curl_exec($ch);
if (curl_errno($ch)) {
echo 'Curl error: ' . curl_error($ch);
}

$information = curl_getinfo($ch);
var_dump($information);

$header = substr($handles, 0, $headerSize);
$body = substr($handles, $headerSize);
}
curl_close($ch);
return $handles;
} catch (Exception $e) {
echo 'Caught exception: ', $e->getMessage(), "\n";
}
    unlink($cookie_file);
}
